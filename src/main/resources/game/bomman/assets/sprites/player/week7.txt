Câu 1:

SELECT e.employeeNumber, e.lastName, e.firstName, e.extension, e.email, e.reportsTo, e.jobTitle, offices.city
FROM employees e
INNER JOIN offices
ON e.officeCode = offices.officeCode;


Câu 2:

SELECT customers.customerName, products.productName
FROM customers
INNER JOIN orders USING(customerNumber)
INNER JOIN orderdetails USING(orderNumber)
INNER JOIN products USING(productCode)
INNER JOIN payments USING(customerNumber);


Câu 3:

SELECT p.*
FROM orderdetails od RIGHT JOIN products p
USING(productCode)
WHERE od.orderNumber IS NULL;


Câu 4:

SELECT o.orderDate, o.requiredDate, o.status, SUM(od.priceEach * od.quantityOrdered) AS 'Total cost'
FROM orders o INNER JOIN orderdetails od
USING(orderNumber)
WHERE MONTH(o.orderDate) = 03 AND YEAR(o.orderDate) = 2005;


Câu 5:

SELECT pl.*, SUM(p.quantityInStock) AS 'Total quantity'
FROM productlines pl
INNER JOIN products p USING(productLine)
GROUP BY productLine
ORDER BY 'Total quantity' DESC;
