Câu 1:
 
SELECT city, COUNT(*) AS 'Number of customers'
FROM customers
GROUP BY city;

Câu 2:

SELECT COUNT(*) AS 'Number of orders'
FROM orders
WHERE EXTRACT(YEAR_MONTH FROM orderDate) = 200503;

Câu 3:

SELECT EXTRACT(YEAR_MONTH FROM orderDate) AS 'Month', COUNT(*) AS 'Number of orders'
FROM orders
GROUP BY Month;

Câu 4:

SELECT orderNumber
FROM orderdetails
GROUP BY orderNumber
ORDER BY SUM(priceEach * quantityOrdered) DESC
LIMIT 10;

Câu 5:

SELECT productLine, SUM(quantityInStock) AS 'Total quantity in stock'
FROM product
GROUP BY productLine;
